        function selectSuggestion(title) {
            document.getElementById('searchInput').value = title;
            document.getElementById('autocompleteDropdown').classList.add('hidden');
            triggerSearch();
        }

        // Custom Key Navigation within instant auto-complete suggestions
        function handleKeyNavigation(event) {
            const dropdown = document.getElementById('autocompleteDropdown');
            if (dropdown.classList.contains('hidden')) {
                if (event.key === 'Enter') {
                    triggerSearch();
                }
                return;
            }

            const items = document.querySelectorAll('.suggestion-item');
            if (event.key === 'ArrowDown') {
                event.preventDefault();
                focusedSuggestionIndex = (focusedSuggestionIndex + 1) % items.length;
                updateFocus(items);
            } else if (event.key === 'ArrowUp') {
                event.preventDefault();
                focusedSuggestionIndex = (focusedSuggestionIndex - 1 + items.length) % items.length;
                updateFocus(items);
            } else if (event.key === 'Enter') {
                event.preventDefault();
                if (focusedSuggestionIndex >= 0 && items[focusedSuggestionIndex]) {
                    items[focusedSuggestionIndex].click();
                } else {
                    triggerSearch();
                }
            } else if (event.key === 'Escape') {
                dropdown.classList.add('hidden');
            }
        }

        function updateFocus(items) {
            items.forEach((item, index) => {
                if (index === focusedSuggestionIndex) {
                    item.classList.add('bg-slate-800', 'border-blue-500');
                    item.scrollIntoView({ block: 'nearest' });
                } else {
                    item.classList.remove('bg-slate-800', 'border-blue-500');
                }
            });
        }

        // Search Action Implementation
        function triggerSearch() {
            const inputVal = document.getElementById('searchInput').value;
            searchQuery = inputVal.trim();
            document.getElementById('autocompleteDropdown').classList.add('hidden');

            // Visual Check for easter egg before continuing standard query
            if (searchQuery.toLowerCase() === 'retro') {
                toggleRetro(true);
            }

            const startTime = performance.now();

            // Core Search Execution logic: Match keywords, tags, title, categories
            const results = MOCK_DATABASE.filter(item => {
                const matchesCategory = currentCategory === 'All' || item.category === currentCategory;
                
                if (!searchQuery) {
                    return matchesCategory; // Show all entries within category if search bar is empty
                }

                const text = searchQuery.toLowerCase();
                const matchesTitle = item.title.toLowerCase().includes(text);
                const matchesDesc = item.description.toLowerCase().includes(text);
                const matchesTags = item.tags.some(tag => tag.toLowerCase().includes(text));

                return matchesCategory && (matchesTitle || matchesDesc || matchesTags);
            });

            const endTime = performance.now();
            const processTime = ((endTime - startTime) * 10).toFixed(3); // Simulated sub-millisecond calculation

            renderResults(results, processTime);
        }

                    return `

                            <div class="space-y-2">
                                <div class="flex items-start justify-between">
                                    <span class="text-[10px] uppercase tracking-wider font-mono px-2.5 py-0.5 rounded-full border ${categoryBadgeColor}">
                                        ${item.category}
                                    </span>
                                    <a href="${item.url}" target="_blank" class="text-slate-500 group-hover:text-blue-400 transition-colors">
                                        <i data-lucide="external-link" class="w-4 h-4"></i>
                                    </a>
                                </div>
                                <h3 class="text-lg font-bold text-slate-100 group-hover:text-blue-300 transition-colors">
                                    <a href="${item.url}" target="_blank">${highlightMatches(item.title)}</a>
                                </h3>
                                <p class="text-sm text-slate-400 leading-relaxed font-light">
                                    ${highlightMatches(item.description)}
                                </p>
                            </div>
                            <div class="flex flex-wrap gap-1.5 mt-4 pt-4 border-t border-slate-800/60">
                                ${item.tags.map(tag => `
                                    <span onclick="setSearch('${tag}')" class="text-[10px] font-mono font-medium text-slate-500 hover:text-blue-400 hover:bg-slate-800 cursor-pointer px-2 py-0.5 bg-slate-950 rounded transition-all">
                                        #${tag}
                                    </span>
                                `).join('')}
                            </div>
                        </div>
                    `;
                }).join('');
                lucide.createIcons();
            }
        }

        // Highlight Searched Term in UI
        function highlightMatches(text) {
            if (!searchQuery) return text;
            const regex = new RegExp(`(${escapeRegExp(searchQuery)})`, 'gi');
            return text.replace(regex, '<mark class="bg-blue-500/30 text-blue-200 border-b border-blue-400 rounded-sm px-0.5">$1</mark>');
        }

        function escapeRegExp(string) {
            return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
        }

        // Clear Search text box helper
        function clearSearch() {
            const input = document.getElementById('searchInput');
            input.value = '';
            searchQuery = '';
            document.getElementById('clearBtn').classList.add('hidden');
            document.getElementById('autocompleteDropdown').classList.add('hidden');
            input.focus();
        }

        // Reset system view back to dashboard home
        function resetToHome() {
            clearSearch();
            const wrapper = document.getElementById('searchWrapper');
            const container = document.getElementById('searchContainer');
            const resultsPanel = document.getElementById('resultsPanel');
            const tagline = document.getElementById('tagline');

            wrapper.classList.add('justify-center');
            wrapper.classList.remove('pt-10', 'justify-start');
            container.classList.add('space-y-8', 'translate-y-0');
            container.classList.remove('space-y-4');
            tagline.classList.remove('hidden');

            resultsPanel.classList.add('hidden');
            setCategory('All');
        }

        // Toggle custom retro green CRT layout
        function toggleRetro(forceMode = false) {
            const isRetro = document.body.classList.toggle('retro-mode', forceMode);
            const btn = document.getElementById('retroBtn');
            if (isRetro) {
                btn.innerHTML = `<i data-lucide="monitor" class="w-3.5 h-3.5"></i> Modern Mode`;
            } else {
                btn.innerHTML = `<i data-lucide="terminal" class="w-3.5 h-3.5"></i> Retro Mode`;
            }
            lucide.createIcons();
        }
    </script>
